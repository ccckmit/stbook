[[code]]
meanRange = function(m, sd) {
	groups = 100
	n = 50
	xmat = matrix(rnorm(n*groups, m, sd), nrow=n, ncol=groups)
	xbar <- apply(xmat, 2, mean)
	xlow = xbar + qnorm(0.025)*1/sqrt(n)
	xhigh = xbar + qnorm(0.975)*1/sqrt(n)
	inRange = count(xlow <= m && xhigh >= m)
	rate = inRange / groups
	return rate
}
